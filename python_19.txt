import pandas as pd

# Create a DataFrame
data = {
    'Student': ['A', 'B', 'C', 'D', 'E'],
    'Score': [65, 82, 91, 47, 76]
}

df = pd.DataFrame(data)

# Filter students who scored more than 70
high_scores = df[df['Score'] > 70]

print("All Students:\n", df)
print("\nStudents with Score > 70:\n", high_scores)
